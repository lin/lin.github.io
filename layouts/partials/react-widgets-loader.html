{{- /* Load React Widget System */ -}} {{- /* Only load if page uses react widgets */ -}} {{- if or (in .Content "react-widget-container") (.Params.useReactWidgets) -}} {{- /* Auto-detect widgets from rendered HTML (look for widget IDs) */ -}} {{- $widgets := slice -}} {{- range $match := findRE `react-widget-([^-]+)-` .Content -}} {{- $component := replaceRE `react-widget-([^-]+)-.*` "$1" $match -}} {{- $widgets = $widgets | append $component -}} {{- end -}} {{- /* Remove duplicates and load each widget */ -}} {{- $uniqueWidgets := slice -}} {{- range $widgets -}} {{- if not (in $uniqueWidgets .) -}} {{- $uniqueWidgets = $uniqueWidgets | append . -}} {{- end -}} {{- end -}} {{- range $uniqueWidgets -}} {{- end -}} {{- end -}}
